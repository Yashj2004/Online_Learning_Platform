create database online_learning_platform;
use online_learning_platform;
-- Courses table
CREATE TABLE Courses (
    course_id INT PRIMARY KEY,
    course_name VARCHAR(255),
    course_description TEXT,
    instructor_id INT,
    start_date DATE,
    end_date DATE,
    duration INT,
    level VARCHAR(50),
    price DECIMAL(10, 2),
    FOREIGN KEY (instructor_id) REFERENCES Instructors(instructor_id)
);

-- Students table
CREATE TABLE Students (
    student_id INT PRIMARY KEY,
    student_name VARCHAR(100),
    email VARCHAR(255),
    date_of_birth DATE,
    country VARCHAR(100),
    registration_date DATE
);

-- Assessments table
CREATE TABLE Assessments (
    assessment_id INT PRIMARY KEY,
    course_id INT,
    assessment_name VARCHAR(255),
    assessment_type VARCHAR(50),
    max_score INT,
    FOREIGN KEY (course_id) REFERENCES Courses(course_id)
);

-- LearningPaths table
CREATE TABLE LearningPaths (
    learning_path_id INT PRIMARY KEY,
    course_id INT,
    learning_path_name VARCHAR(255),
    FOREIGN KEY (course_id) REFERENCES Courses(course_id)
);

-- StudentProgress table
CREATE TABLE StudentProgress (
    progress_id INT PRIMARY KEY,
    student_id INT,
    course_id INT,
    assessment_id INT,
    completion_date DATE,
    score INT,
    FOREIGN KEY (student_id) REFERENCES Students(student_id),
    FOREIGN KEY (course_id) REFERENCES Courses(course_id),
    FOREIGN KEY (assessment_id) REFERENCES Assessments(assessment_id)
);

-- Instructors table
CREATE TABLE Instructors (
    instructor_id INT PRIMARY KEY,
    instructor_name VARCHAR(100),
    email VARCHAR(255),
    bio TEXT,
    expertise_area VARCHAR(255)
);

show tables;

-- Generating 1000 rows of random data
INSERT INTO StudentProgress (progress_id, student_id, course_id, assessment_id, completion_date, score)
SELECT 
    ROW_NUMBER() OVER (ORDER BY (SELECT NULL)) AS progress_id,
    FLOOR(RAND() * 200) + 1 AS student_id,
    FLOOR(RAND() * 200) + 1 AS course_id,
    FLOOR(RAND() * 400) + 1 AS assessment_id,
    DATE_ADD('2024-01-01', INTERVAL FLOOR(RAND() * 150) DAY) AS completion_date, -- Random date between 2024-01-01 and 2024-05-30
    FLOOR(RAND() * 101) AS score -- Random score between 0 and 100
FROM
    INFORMATION_SCHEMA.TABLES T1,
    INFORMATION_SCHEMA.TABLES T2
LIMIT 1000;
